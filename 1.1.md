def main():
    # 1. Десятичное число
    a = 42

    # 2. Двоичное число (префикс 0b)
    b = 0b101010

    # 3. Восьмеричное число (префикс 0o)
    c = 0o52

    # 4. Шестнадцатеричное число (префикс 0x)
    d = 0x2A

    # 5. Строка, преобразованная в целое число
    e = int("42")

    # 6. Список с одним элементом 42
    f = [42][0]

    # 7. Кортеж с одним элементом 42
    g = (42,)[0]

    # 8. Множество с одним элементом 42
    h = {42}.pop()

    # 9. Словарь, где ключ — строка, а значение — 42
    i = {"answer": 42}["answer"]

    # 10. Булевое значение True, преобразованное в целое число (True = 1, False = 0)
    j = int(True) * 42

    # Проверка, что все переменные равны 42
    if a == b == c == d == e == f == g == h == i == j == 42:
        print("Все значения равны 42!")
    else:
        print("Что-то пошло не так...")

    # Вывод всех значений для наглядности
    print(f"a = {a}")
    print(f"b = {b}")
    print(f"c = {c}")
    print(f"d = {d}")
    print(f"e = {e}")
    print(f"f = {f}")
    print(f"g = {g}")
    print(f"h = {h}")
    print(f"i = {i}")
    print(f"j = {j}")

# Запуск программы
if __name__ == "__main__":
    main()
